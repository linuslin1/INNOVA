%%R
weeklydata = function(x){
    a = matrix(x$Adj.Close,7)
    colmeans = colMeans(a)
    wweek = c(1:length(colmeans))
    dataf = data.frame(wweek, colmeans)
    colnames(dataf) = c("Week", "Adj Close")
    dataf}
